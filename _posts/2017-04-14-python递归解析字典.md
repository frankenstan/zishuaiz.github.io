python 递归解析字典的key，并追加到list中


```python
def print_dict(dictionary, ret, parent_key=''):
    if isinstance(dictionary, dict):
        for key, value in dictionary.items():
            if isinstance(value, dict):
                ret.append(parent_key + '.' + key)
                print(parent_key + '.' + key)
                print_dict(value, ret, parent_key + '.' + key)
            elif isinstance(value, list):
                for i, v in enumerate(value):
                    print_dict(v, ret, parent_key + '.' + key + '[' + str(i) + ']')
                    ret.append(parent_key + '.' + key)
            else:
                print(parent_key + '.' + key)
                ret.append(parent_key + '.' + key)
    elif isinstance(dictionary, list):
        for i, v in enumerate(dictionary):
            print_dict(v, ret, parent_key + '.' + '[' + str(i) + ']')
```
